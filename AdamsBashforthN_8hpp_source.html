138  /*!
139  * An explanation of the computation being performed by this
140  * function:
141  * \f$\newcommand\tL{t^L}\newcommand\tR{t^R}\newcommand\tU{\tilde{t}\!}
142  * \newcommand\mat{\mathbf}\f$
143  *
144  * Suppose the local and remote sides of the interface are evaluated
145  * at times \f$\ldots, \tL_{-1}, \tL_0, \tL_1, \ldots\f$ and
146  * \f$\ldots, \tR_{-1}, \tR_0, \tR_1, \ldots\f$, respectively, with
147  * the starting location of the numbering arbitrary in each case.
148  * Let the step we wish to calculate the effect of be the step from
149  * \f$\tL_{m_S}\f$ to \f$\tL_{m_S+1}\f$. We call the sequence
150  * produced from the union of the local and remote time sequences
151  * \f$\ldots, \tU_{-1}, \tU_0, \tU_1, \ldots\f$. For example, one
152  * possible sequence of times is:
153  * \f{equation}
154  * \begin{aligned}
155  * \text{Local side:} \\ \text{Union times:} \\ \text{Remote side:}
156  * \end{aligned}
157  * \cdots
158  * \begin{gathered}
159  * \, \\ \tU_1 \\ \tR_5
160  * \end{gathered}
161  * \leftarrow \Delta \tU_1 \rightarrow
162  * \begin{gathered}
163  * \tL_4 \\ \tU_2 \\ \,
164  * \end{gathered}
165  * \leftarrow \Delta \tU_2 \rightarrow
166  * \begin{gathered}
167  * \, \\ \tU_3 \\ \tR_6
168  * \end{gathered}
169  * \leftarrow \Delta \tU_3 \rightarrow
170  * \begin{gathered}
171  * \, \\ \tU_4 \\ \tR_7
172  * \end{gathered}
173  * \leftarrow \Delta \tU_4 \rightarrow
174  * \begin{gathered}
175  * \tL_5 \\ \tU_5 \\ \,
176  * \end{gathered}
177  * \cdots
178  * \f}
179  * We call the indices of the step's start and end times in the
180  * union time sequence \f$n_S\f$ and \f$n_E\f$, respectively. We
181  * define \f$n^L_m\f$ to be the union-time index corresponding to
182  * \f$\tL_m\f$ and \f$m^L_n\f$ to be the index of the last local
183  * time not later than \f$\tU_n\f$ and similarly for the remote
184  * side. So for the above example, \f$n^L_4 = 2\f$ and \f$m^R_2 =
185  * 5\f$, and if we wish to compute the step from \f$\tL_4\f$ to
186  * \f$\tL_5\f$ we would have \f$m_S = 4\f$, \f$n_S = 2\f$, and
187  * \f$n_E = 5\f$.
188  *
189  * If we wish to evaluate the change over this step to \f$k\f$th
190  * order, we can write the change in the value as a linear
191  * combination of the values of the coupling between the elements at
192  * unequal times:
193  * \f{equation}
194  * \mat{F}_{m_S} =
195  * \mspace{-10mu}
196  * \sum_{q^L = m_S-(k-1)}^{m_S}
197  * \,
198  * \sum_{q^R = m^R_{n_S}-(k-1)}^{m^R_{n_E-1}}
199  * \mspace{-10mu}
200  * \mat{D}_{q^Lq^R}
201  * I_{q^Lq^R},
202  * \f}
203  * where \f$\mat{D}_{q^Lq^R}\f$ is the coupling function evaluated
204  * between data from \f$\tL_{q^L}\f$ and \f$\tR_{q^R}\f$. The
205  * coefficients can be written as the sum of three terms,
206  * \f{equation}
207  * I_{q^Lq^R} = I^E_{q^Lq^R} + I^R_{q^Lq^R} + I^L_{q^Lq^R},
208  * \f}
209  * which can be interpreted as a contribution from equal-time
210  * evaluations and contributions related to the remote and local
211  * evaluation times. These are given by
212  * \f{align}
213  * I^E_{q^Lq^R} &=
214  * \mspace{-10mu}
215  * \sum_{n=n_S}^{\min\left\{n_E, n^L+k\right\}-1}
216  * \mspace{-10mu}
217  * \tilde{\alpha}_{n,n-n^L} \Delta \tU_n
218  * &&\text{if $\tL_{q^L} = \tR_{q^R}$, otherwise 0}
219  * \\
220  * I^R_{q^Lq^R} &=
221  * \ell_{q^L - m_S + k}\!\left(
222  * \tU_{n^R}; \tL_{m_S - (k-1)}, \ldots, \tL_{m_S}\right)
223  * \mspace{-10mu}
224  * \sum_{n=\max\left\{n_S, n^R\right\}}
225  * ^{\min\left\{n_E, n^R+k\right\}-1}
226  * \mspace{-10mu}
227  * \tilde{\alpha}_{n,n-n^R} \Delta \tU_n
228  * &&\text{if $\tR_{q^R}$ is not in $\{\tL_{\vphantom{|}\cdots}\}$,
229  * otherwise 0}
230  * \\
231  * I^L_{q^Lq^R} &=
232  * \mspace{-10mu}
233  * \sum_{n=\max\left\{n_S, n^R\right\}}
234  * ^{\min\left\{n_E, n^L+k, n^R_{q^R+k}\right\}-1}
235  * \mspace{-10mu}
236  * \ell_{q^R - m^R_n + k}\!\left(\tU_{n^L};
237  * \tR_{m^R_n - (k-1)}, \ldots, \tR_{m^R_n}\right)
238  * \tilde{\alpha}_{n,n-n^L} \Delta \tU_n
239  * &&\text{if $\tL_{q^L}$ is not in $\{\tR_{\vphantom{|}\cdots}\}$,
240  * otherwise 0,}
241  * \f}
242  * where for brevity we write \f$n^L = n^L_{q^L}\f$ and \f$n^R =
243  * n^R_{q^R}\f$, and where \f$\ell_a(t; x_1, \ldots, x_k)\f$ a
244  * Lagrange interpolating polynomial and \f$\tilde{\alpha}_{nj}\f$
245  * is the \f$j\f$th coefficient for an Adams-Bashforth step over the
246  * union times from step \f$n\f$ to step \f$n+1\f$.
247  */

419 template <typename UpdateVars, typename Vars, typename DerivVars,
420  typename Delta>
421 void AdamsBashforthN::update_u_impl(const gsl::not_null<UpdateVars*> u,
422  const History<Vars, DerivVars>& history,
423  const Delta& time_step,
424  const size_t order) const noexcept {
425  ASSERT(
426  history.size() > 0,
427  "Cannot meaningfully update the evolved variables with an empty history");
428  ASSERT(order <= order_,
429  "Requested integration order higher than integrator order");
430 
431  const auto history_start =
432  history.end() -
433  static_cast<typename History<Vars, DerivVars>::difference_type>(order);
434  const auto coefficients =
435  get_coefficients(history_start, history.end(), time_step);
436 
437  *u = history.most_recent_value();
438  auto coefficient = coefficients.rbegin();
439  for (auto history_entry = history_start;
440  history_entry != history.end();
441  ++history_entry, ++coefficient) {
442  *u += time_step.value() * *coefficient * history_entry.derivative();
443  }
444 }

446 template <typename LocalVars, typename RemoteVars, typename Coupling>
447 std::result_of_t<const Coupling&(LocalVars, RemoteVars)>
448 AdamsBashforthN::compute_boundary_delta(
449  const Coupling& coupling,
450  const gsl::not_null<BoundaryHistoryType<LocalVars, RemoteVars, Coupling>*>
451  history,
452  const TimeDelta& time_step) const noexcept {
453  const auto signed_order =
454  static_cast<typename decltype(history->local_end())::difference_type>(
455  history->integration_order());
456 
457  ASSERT(history->local_size() >= history->integration_order(),
458  "Insufficient data to take an order-" << history->integration_order()
459  << " step. Have " << history->local_size() << " times, need "
460  << history->integration_order());
461  history->local_mark_unneeded(history->local_end() - signed_order);
462 
463  if (std::equal(history->local_begin(), history->local_end(),
464  history->remote_end() - signed_order)) {
465  // GTS
466  ASSERT(history->remote_size() >= history->integration_order(),
467  "Insufficient data to take an order-" << history->integration_order()
468  << " step. Have " << history->remote_size() << " times, need "
469  << history->integration_order());
470  history->remote_mark_unneeded(history->remote_end() - signed_order);
471  } else {
472  const auto remote_step_for_step_start =
473  std::upper_bound(history->remote_begin(), history->remote_end(),
474  *(history->local_end() - 1),
475  evolution_less<Time>{time_step.is_positive()});
476  ASSERT(remote_step_for_step_start - history->remote_begin() >= signed_order,
477  "Insufficient data to take an order-" << history->integration_order()
478  << " step. Have "
479  << remote_step_for_step_start - history->remote_begin()
480  << " times before the step, need " << history->integration_order());
481  history->remote_mark_unneeded(remote_step_for_step_start - signed_order);
482  }
483 
484  return boundary_impl(coupling, *history,
485  *(history->local_end() - 1) + time_step);
486 }

497 template <typename LocalVars, typename RemoteVars, typename Coupling,
498  typename TimeType>
499 std::result_of_t<const Coupling&(LocalVars, RemoteVars)>
500 AdamsBashforthN::boundary_impl(
501  const Coupling& coupling,
502  const BoundaryHistoryType<LocalVars, RemoteVars, Coupling>& history,
503  const TimeType& end_time) const noexcept {
504  // Might be different from order_ during self-start.
505  const auto current_order = history.integration_order();
506 
507  ASSERT(current_order <= order_,
508  "Local history is too long for target order (" << current_order
509  << " should not exceed " << order_ << ")");
510  ASSERT(history.remote_size() >= current_order,
511  "Remote history is too short (" << history.remote_size()
512  << " should be at least " << current_order << ")");
513 
514  // Avoid billions of casts
515  const auto order_s = static_cast<typename BoundaryHistoryType<
516  LocalVars, RemoteVars, Coupling>::remote_iterator::difference_type>(
517  current_order);
518 
519  // Start and end of the step we are trying to take
520  const Time start_time = *(history.local_end() - 1);
521  const auto time_step = end_time - start_time;
522 
523  // Result variable. We evaluate the coupling only for the
524  // structure. This evaluation may be expensive, but by choosing the
525  // most recent times on both sides we should guarantee that it is a
526  // result we need later, so this will serve to get it into the
527  // coupling cache so we don't have to compute it when we actually use it.
528  auto accumulated_change =
529  make_with_value<std::result_of_t<const Coupling&(LocalVars, RemoteVars)>>(
530  history.coupling(coupling, history.local_end() - 1,
531  history.remote_end() - 1),
532  0.);
533 
534  // We define the local_begin and remote_begin variables as the start
535  // of the part of the history relevant to this calculation.
536  // Boundary history cleanup happens immediately before the step, but
537  // boundary dense output happens before that, so there may be data
538  // left over that was needed for the previous step and has not been
539  // cleaned out yet.
540  const auto local_begin = history.local_end() - order_s;
541 
542  if (std::equal(local_begin, history.local_end(),
543  history.remote_end() - order_s)) {
544  // No local time-stepping going on.
545  const auto coefficients =
546  get_coefficients(local_begin, history.local_end(), time_step);
547 
548  auto local_it = local_begin;
549  auto remote_it = history.remote_end() - order_s;
550  for (auto coefficients_it = coefficients.rbegin();
551  coefficients_it != coefficients.rend();
552  ++coefficients_it, ++local_it, ++remote_it) {
553  accumulated_change +=
554  *coefficients_it * history.coupling(coupling, local_it, remote_it);
555  }
556  accumulated_change *= time_step.value();
557 
558  return accumulated_change;
559  }
560 
561  ASSERT(current_order == order_,
562  "Cannot perform local time-stepping while self-starting.");
563 
564  const evolution_less<> less{time_step.is_positive()};
565  const auto remote_begin =
566  std::upper_bound(history.remote_begin(), history.remote_end(),
567  start_time, less) -
568  order_s;
569 
570  ASSERT(std::is_sorted(local_begin, history.local_end(), less),
571  "Local history not in order");
572  ASSERT(std::is_sorted(remote_begin, history.remote_end(), less),
573  "Remote history not in order");
574  ASSERT(not less(start_time, *(remote_begin + (order_s - 1))),
575  "Remote history does not extend far enough back");
576  ASSERT(less(*(history.remote_end() - 1), end_time),
577  "Please supply only older data: " << *(history.remote_end() - 1)
578  << " is not before " << end_time);
579 
580  // Union of times of all step boundaries on any side.
581  const auto union_times = [&history, &local_begin, &remote_begin,
582  &less]() noexcept {
583  std::vector<Time> ret;
584  ret.reserve(history.local_size() + history.remote_size());
585  std::set_union(local_begin, history.local_end(), remote_begin,
586  history.remote_end(), std::back_inserter(ret), less);
587  return ret;
588  }();
589 
590  using UnionIter = typename decltype(union_times)::const_iterator;
591 
592  // Find the union times iterator for a given time.
593  const auto union_step = [&union_times, &less](const Time& t) noexcept {
594  return std::lower_bound(union_times.cbegin(), union_times.cend(), t, less);
595  };
596 
597  // The union time index for the step start.
598  const auto union_step_start = union_step(start_time);
599 
600  // min(union_times.end(), it + order_s) except being careful not
601  // to create out-of-range iterators.
602  const auto advance_within_step =
603  [order_s, &union_times](const UnionIter& it) noexcept {
604  return union_times.end() - it >
605  static_cast<typename decltype(union_times)::difference_type>(
606  order_s)
607  ? it + static_cast<typename decltype(
608  union_times)::difference_type>(order_s)
609  : union_times.end();
610  };
611 
612  // Calculating the Adams-Bashforth coefficients is somewhat
613  // expensive, so we cache them. ab_coefs(it, step) returns the
614  // coefficients used to step from *it to *it + step.
615  auto ab_coefs = make_overloader(
616  make_cached_function<std::tuple<UnionIter, TimeDelta>,
617  std::map>([order_s](
618  const std::tuple<UnionIter, TimeDelta>& args) noexcept {
619  return get_coefficients(
620  std::get<0>(args) -
621  static_cast<typename UnionIter::difference_type>(order_s - 1),
622  std::get<0>(args) + 1, std::get<1>(args));
623  }),
624  make_cached_function<std::tuple<UnionIter, ApproximateTimeDelta>,
625  std::map>([order_s](
626  const std::tuple<UnionIter, ApproximateTimeDelta>& args) noexcept {
627  return get_coefficients(
628  std::get<0>(args) -
629  static_cast<typename UnionIter::difference_type>(order_s - 1),
630  std::get<0>(args) + 1, std::get<1>(args));
631  }));
632 
633  // The value of the coefficient of `evaluation_step` when doing
634  // a standard Adams-Bashforth integration over the union times
635  // from `step` to `step + 1`.
636  const auto base_summand = [&ab_coefs, &end_time, &union_times](
637  const UnionIter& step, const UnionIter& evaluation_step) noexcept {
638  if (step + 1 != union_times.end()) {
639  const TimeDelta step_size = *(step + 1) - *step;
640  return step_size.value() *
641  ab_coefs(std::make_tuple(
642  step, step_size))[static_cast<size_t>(step - evaluation_step)];
643  } else {
644  const auto step_size = end_time - *step;
645  return step_size.value() *
646  ab_coefs(std::make_tuple(
647  step, step_size))[static_cast<size_t>(step - evaluation_step)];
648  }
649  };
650 
651  for (auto local_evaluation_step = local_begin;
652  local_evaluation_step != history.local_end();
653  ++local_evaluation_step) {
654  const auto union_local_evaluation_step = union_step(*local_evaluation_step);
655  for (auto remote_evaluation_step = remote_begin;
656  remote_evaluation_step != history.remote_end();
657  ++remote_evaluation_step) {
658  double deriv_coef = 0.;
659 
660  if (*local_evaluation_step == *remote_evaluation_step) {
661  // The two elements stepped at the same time. This gives a
662  // standard Adams-Bashforth contribution to each segment
663  // making up the current step.
664  const auto union_step_upper_bound =
665  advance_within_step(union_local_evaluation_step);
666  for (auto step = union_step_start;
667  step < union_step_upper_bound;
668  ++step) {
669  deriv_coef += base_summand(step, union_local_evaluation_step);
670  }
671  } else {
672  // In this block we consider a coupling evaluation that is not
673  // performed at equal times on the two sides of the mortar.
674 
675  // Makes an iterator with a map to give time as a double.
676  const auto make_lagrange_iterator = [](const auto& it) noexcept {
677  return boost::make_transform_iterator(
678  it, [](const Time& t) noexcept { return t.value(); });
679  };
680 
681  const auto union_remote_evaluation_step =
682  union_step(*remote_evaluation_step);
683  const auto union_step_lower_bound =
684  std::max(union_step_start, union_remote_evaluation_step);
685 
686  // Compute the contribution to an interpolation over the local
687  // times to `remote_evaluation_step->value()`, which we will
688  // use as the coupling value for that time. If there is an
689  // actual evaluation at that time then skip this because the
690  // Lagrange polynomial will be zero.
691  if (not std::binary_search(local_begin, history.local_end(),
692  *remote_evaluation_step, less)) {
693  const auto union_step_upper_bound =
694  advance_within_step(union_remote_evaluation_step);
695  for (auto step = union_step_lower_bound;
696  step < union_step_upper_bound;
697  ++step) {
698  deriv_coef += base_summand(step, union_remote_evaluation_step);
699  }
700  deriv_coef *=
701  lagrange_polynomial(make_lagrange_iterator(local_evaluation_step),
702  remote_evaluation_step->value(),
703  make_lagrange_iterator(local_begin),
704  make_lagrange_iterator(history.local_end()));
705  }
706 
707  // Same qualitative calculation as the previous block, but
708  // interpolating over the remote times. This case is somewhat
709  // more complicated because the latest remote time that can be
710  // used varies for the different segments making up the step.
711  if (not std::binary_search(remote_begin, history.remote_end(),
712  *local_evaluation_step, less)) {
713  auto union_step_upper_bound =
714  advance_within_step(union_local_evaluation_step);
715  if (history.remote_end() - remote_evaluation_step > order_s) {
716  union_step_upper_bound = std::min(
717  union_step_upper_bound,
718  union_step(*(remote_evaluation_step + order_s)));
719  }
720 
721  auto control_points = make_lagrange_iterator(
722  remote_evaluation_step - remote_begin >= order_s
723  ? remote_evaluation_step - (order_s - 1)
724  : remote_begin);
725  for (auto step = union_step_lower_bound;
726  step < union_step_upper_bound;
727  ++step, ++control_points) {
728  deriv_coef +=
729  base_summand(step, union_local_evaluation_step) *
730  lagrange_polynomial(
731  make_lagrange_iterator(remote_evaluation_step),
732  local_evaluation_step->value(), control_points,
733  control_points +
734  static_cast<typename decltype(
735  control_points)::difference_type>(order_s));
736  }
737  }
738  }
739 
740  if (deriv_coef != 0.) {
741  // Skip the (potentially expensive) coupling calculation if
742  // the coefficient is zero.
743  accumulated_change +=
744  deriv_coef * history.coupling(coupling, local_evaluation_step,
745  remote_evaluation_step);
746  }
747  } // for remote_evaluation_step
748  } // for local_evaluation_step
749 
750  return accumulated_change;
751 }

753 template <typename Vars, typename DerivVars>
754 bool AdamsBashforthN::can_change_step_size(
755  const TimeStepId& time_id,
756  const TimeSteppers::History<Vars, DerivVars>& history) const noexcept {
757  // We need to forbid local time-stepping before initialization is
758  // complete. The self-start procedure itself should never consider
759  // changing the step size, but we need to wait during the main
760  // evolution until the self-start history has been replaced with
761  // "real" values.
762  const evolution_less<Time> less{time_id.time_runs_forward()};
763  return not ::SelfStart::is_self_starting(time_id) and
764  (history.size() == 0 or
765  (less(history.back(), time_id.step_time()) and
766  std::is_sorted(history.begin(), history.end(), less)));
767 }

769 template <typename Iterator, typename Delta>
770 std::vector<double> AdamsBashforthN::get_coefficients(
771  const Iterator& times_begin, const Iterator& times_end,
772  const Delta& step) noexcept {
773  if (times_begin == times_end) {
774  return {};
775  }
776  std::vector<double> steps;
777  // This may be slightly more space than we need, but we can't get
778  // the exact amount without iterating through the iterators, which
779  // is not necessarily cheap depending on the iterator type.
780  steps.reserve(maximum_order);
781  for (auto t = times_begin; std::next(t) != times_end; ++t) {
782  steps.push_back((*std::next(t) - *t).value());
783  }
784  steps.push_back(step.value());
785  return get_coefficients_impl(steps);
786 }
