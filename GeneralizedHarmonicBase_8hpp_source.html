286  template <typename... Tags>
287  static Phase determine_next_phase(
288  const gsl::not_null<tuples::TaggedTuple<Tags...>*>
289  phase_change_decision_data,
290  const Phase& current_phase,
291  const Parallel::CProxy_GlobalCache<derived_metavars>&
292  cache_proxy) noexcept {
293  const auto next_phase = PhaseControl::arbitrate_phase_change<phase_changes>(
294  phase_change_decision_data, current_phase,
295  *(cache_proxy.ckLocalBranch()));
296  if (next_phase.has_value()) {
297  return next_phase.value();
298  }
299  switch (current_phase) {
300  case Phase::Initialization:
301  return evolution::is_numeric_initial_data_v<initial_data>
302  ? Phase::RegisterWithElementDataReader
303  : Phase::InitializeInitialDataDependentQuantities;
304  case Phase::RegisterWithElementDataReader:
305  return Phase::ImportInitialData;
306  case Phase::ImportInitialData:
307  return Phase::InitializeInitialDataDependentQuantities;
308  case Phase::InitializeInitialDataDependentQuantities:
309  return Phase::InitializeTimeStepperHistory;
310  case Phase::InitializeTimeStepperHistory:
311  return Phase::Register;
312  case Phase::Register:
313  return Phase::Evolve;
314  case Phase::Evolve:
315  return Phase::Exit;
316  case Phase::Exit:
317  ERROR(
318  "Should never call determine_next_phase with the current phase "
319  "being 'Exit'");
320  default:
321  ERROR(
322  "Unknown type of phase. Did you static_cast<Phase> an integral "
323  "value?");
324  }
325  }
