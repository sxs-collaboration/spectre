65 /*!
66  * \ingroup CharmExtensionsGroup
67  * \brief Returns the template parameter as a `std::string`
68  *
69  * Uses the __PRETTY_FUNCTION__ compiler intrinsic to extract the template
70  * parameter names in the same form that Charm++ uses to register entry methods.
71  * This is used by the generated Singleton, Array, Group and Nodegroup headers,
72  * as well as in CharmMain.tpp.
73  */
74 template <class... Args>
75 std::string get_template_parameters_as_string() {
76  std::string function_name(static_cast<char const*>(__PRETTY_FUNCTION__));
77  std::string template_params =
78  function_name.substr(function_name.find(std::string("Args = ")) + 8);
79  template_params.erase(template_params.end() - 2, template_params.end());
80  size_t pos = 0;
81  while ((pos = template_params.find(" >")) != std::string::npos) {
82  template_params.replace(pos, 1, ">");
83  template_params.erase(pos + 1, 1);
84  }
85  pos = 0;
86  while ((pos = template_params.find(", ", pos)) != std::string::npos) {
87  template_params.erase(pos + 1, 1);
88  }
89  pos = 0;
90  while ((pos = template_params.find('>', pos + 2)) != std::string::npos) {
91  template_params.replace(pos, 1, " >");
92  }
93  std::replace(template_params.begin(), template_params.end(), '%', '>');
94  // GCC's __PRETTY_FUNCTION__ adds the return type at the end, so we remove it.
95  if (template_params.find('}') != std::string::npos) {
96  template_params.erase(template_params.find('}'), template_params.size());
97  }
98  return template_params;
99 }

101 /*!
102  * \ingroup CharmExtensionsGroup
103  * \brief The base class used for automatic registration of entry methods.
104  *
105  * Entry methods are automatically registered by building a list of the entry
106  * methods that need to be registered in the `charm_register_list`. All entry
107  * methods in the list are later registered in the
108  * `register_parallel_components` function, at which point the list is also
109  * deleted.
110  *
111  * The reason for using an abstract base class mechanism is that we need to be
112  * able to register entry method templates. The derived classes keep track of
113  * the template parameters and override the `register_with_charm` function.
114  * The result is that there must be one derived class template for each entry
115  * method template, but since we only have a few entry method templates this is
116  * not an issue.
117  */
118 struct RegistrationHelper {
119  RegistrationHelper() = default;
120  RegistrationHelper(const RegistrationHelper&) = default;
121  RegistrationHelper& operator=(const RegistrationHelper&) = default;
122  RegistrationHelper(RegistrationHelper&&) = default;
123  RegistrationHelper& operator=(RegistrationHelper&&) = default;
124  virtual ~RegistrationHelper() = default;
125 
126  virtual void register_with_charm() const noexcept = 0;
127  virtual std::string name() const noexcept = 0;
128  virtual bool is_registering_chare() const noexcept { return false; };
129 };

388  // in a non-const Key type, which is different from what

390  // difference leads to issues with function registration with Charm++ because
391  // when registering `receive_data` methods we register the `inbox_type`'s

393  // passed to `receive_data`.

625 /*!
626  * \ingroup CharmExtensionsGroup
627  * \brief Function that adds a pointer to a specific derived class to the
628  * `charm_register_list`
629  *
630  * Used to initialize the `registrar` bool of derived classes of
631  * `RegistrationHelper`. When the function is invoked it appends the derived
632  * class to the `charm_register_list`.
633  *
634  * \note The reason for not using a `std::vector` is that this did not behave
635  * correctly when calling `push_back`. Specifically, the final vector was always

644  // necessary)
645  auto* const t = // NOLINT
646  new std::unique_ptr<RegistrationHelper>[charm_register_list_capacity +
647  10];
648  for (size_t i = 0; i < charm_register_list_capacity; ++i) {
649  // clang-tidy: do not use pointer arithmetic
650  t[i] = std::move(charm_register_list[i]); // NOLINT

653  // necessary)

660  charm_register_list[charm_register_list_size - 1] = // NOLINT
