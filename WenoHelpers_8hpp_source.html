28 // Compute the WENO weighted reconstruction of a DataVector, see e.g.,
29 // Eq. 4.3 of Zhu2016. This is fairly standard, though different references can
30 // differ in their choice of oscillation/smoothness indicator. The
31 // `DerivativeWeight` enum specifies the relative weight of each derivative
32 // term when computing the oscillation indicator.
33 //
34 // The reconstruction modifies the DataVector in `local_polynomial` by adding
35 // to it a weighted combination of one or more neighbor contributions, passed
36 // in as several DataVectors in `neighbor_polynommials`. Each neighbor
37 // polynomial must have the same mean as the local polynomial; this is checked
38 // with an ASSERT.
39 template <size_t VolumeDim>
40 void reconstruct_from_weighted_sum(
41  gsl::not_null<DataVector*> local_polynomial, double neighbor_linear_weight,
42  DerivativeWeight derivative_weight, const Mesh<VolumeDim>& mesh,
43  const std::unordered_map<
44  std::pair<Direction<VolumeDim>, ElementId<VolumeDim>>, DataVector,
45  boost::hash<std::pair<Direction<VolumeDim>, ElementId<VolumeDim>>>>&
46  neighbor_polynomials) noexcept;
