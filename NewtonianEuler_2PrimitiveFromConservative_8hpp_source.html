37 /*!
38  * \brief Compute the primitive variables from the conservative variables.
39  *
40  * \f{align*}
41  * v^i &= \frac{S^i}{\rho} \\
42  * \epsilon &= \frac{e}{\rho} - \frac{1}{2}\frac{S^2}{\rho^2}
43  * \f}
44  *
45  * where \f$v^i\f$ is the velocity, \f$\epsilon\f$ is the specific
46  * internal energy, \f$e\f$ is the energy density, \f$\rho\f$
47  * is the mass density, \f$S^i\f$ is the momentum density, and
48  * \f$S^2\f$ is the momentum density squared.
49  *
50  * This routine also returns the mass density as a primitive, and the pressure
51  * from a generic equation of state \f$p = p(\rho, \epsilon)\f$.
52  */
53 template <size_t Dim, size_t ThermodynamicDim>
54 struct PrimitiveFromConservative {
55  using return_tags =
56  tmpl::list<Tags::MassDensity<DataVector>, Tags::Velocity<DataVector, Dim>,
57  Tags::SpecificInternalEnergy<DataVector>,
58  Tags::Pressure<DataVector>>;
59 
60  using argument_tags =
61  tmpl::list<Tags::MassDensityCons, Tags::MomentumDensity<Dim>,
62  Tags::EnergyDensity, hydro::Tags::EquationOfStateBase>;
63 
64  static void apply(
65  gsl::not_null<Scalar<DataVector>*> mass_density,
66  gsl::not_null<tnsr::I<DataVector, Dim>*> velocity,
67  gsl::not_null<Scalar<DataVector>*> specific_internal_energy,
68  gsl::not_null<Scalar<DataVector>*> pressure,
69  const Scalar<DataVector>& mass_density_cons,
70  const tnsr::I<DataVector, Dim>& momentum_density,
71  const Scalar<DataVector>& energy_density,
72  const EquationsOfState::EquationOfState<false, ThermodynamicDim>&
73  equation_of_state) noexcept;
74 };
75 } // namespace NewtonianEuler
