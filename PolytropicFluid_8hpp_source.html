28 namespace EquationsOfState {
29 /*!
30  * \ingroup EquationsOfStateGroup
31  * \brief Equation of state for a polytropic fluid
32  *
33  * A polytropic equation of state \f$p=K\rho^{\Gamma}\f$ where \f$K\f$ is the
34  * polytropic constant and \f$\Gamma\f$ is the polytropic exponent. The
35  * polytropic exponent is related to the polytropic index \f$N_p\f$ by
36  * \f$N_p=1/(\Gamma-1)\f$.
37  */
38 template <bool IsRelativistic>
39 class PolytropicFluid : public EquationOfState<IsRelativistic, 1> {
40  public:
41  static constexpr size_t thermodynamic_dim = 1;
42  static constexpr bool is_relativistic = IsRelativistic;
43 
44  struct PolytropicConstant {
45  using type = double;
46  static constexpr Options::String help = {"Polytropic constant K"};
47  static double lower_bound() noexcept { return 0.0; }
48  };
49 
50  struct PolytropicExponent {
51  using type = double;
52  static constexpr Options::String help = {"Polytropic exponent Gamma"};
53  static double lower_bound() noexcept { return 1.0; }
54  };
55 
56  static constexpr Options::String help = {
57  "A polytropic fluid equation of state.\n"
58  "The pressure is related to the rest mass density by p = K rho ^ Gamma, "
59  "where p is the pressure, rho is the rest mass density, K is the "
60  "polytropic constant, and Gamma is the polytropic exponent. The "
61  "polytropic index N is defined as Gamma = 1 + 1 / N."};
62 
63  using options = tmpl::list<PolytropicConstant, PolytropicExponent>;
64 
65  PolytropicFluid() = default;
66  PolytropicFluid(const PolytropicFluid&) = default;
67  PolytropicFluid& operator=(const PolytropicFluid&) = default;
68  PolytropicFluid(PolytropicFluid&&) = default;
69  PolytropicFluid& operator=(PolytropicFluid&&) = default;
70  ~PolytropicFluid() override = default;
71 
72  PolytropicFluid(double polytropic_constant,
73  double polytropic_exponent) noexcept;
74 
75  EQUATION_OF_STATE_FORWARD_DECLARE_MEMBERS(PolytropicFluid, 1)
76 
77  WRAPPED_PUPable_decl_base_template( // NOLINT
78  SINGLE_ARG(EquationOfState<IsRelativistic, 1>), PolytropicFluid);
79 
80  /// The lower bound of the rest mass density that is valid for this EOS
81  double rest_mass_density_lower_bound() const noexcept override { return 0.0; }
82 
83  /// The upper bound of the rest mass density that is valid for this EOS
84  double rest_mass_density_upper_bound() const noexcept override;
85 
86  /// The lower bound of the specific internal energy that is valid for this EOS
87  /// at the given rest mass density \f$\rho\f$
88  double specific_internal_energy_lower_bound(
89  const double /* rest_mass_density */) const noexcept override {
90  return 0.0;
91  }
92 
93  /// The upper bound of the specific internal energy that is valid for this EOS
94  /// at the given rest mass density \f$\rho\f$
95  double specific_internal_energy_upper_bound(
96  const double /* rest_mass_density */) const noexcept override {
97  return std::numeric_limits<double>::max();
98  }
99 
100  /// The lower bound of the specific enthalpy that is valid for this EOS
101  double specific_enthalpy_lower_bound() const noexcept override {
102  return IsRelativistic ? 1.0 : 0.0;
103  }
104 
105  private:
106  EQUATION_OF_STATE_FORWARD_DECLARE_MEMBER_IMPLS(1)
107 
108  double polytropic_constant_ = std::numeric_limits<double>::signaling_NaN();
109  double polytropic_exponent_ = std::numeric_limits<double>::signaling_NaN();
110 };
