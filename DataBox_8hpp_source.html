567 // that depended upon the mutated items. It starts with
568 // TagsOfImmutableItemsToReset as the tags of the immutable items that directly
569 // depended upon the mutated items. This function resets the compute items in
570 // TagsOfImmutableItemsToReset and then constructs the list of immutable items
571 // that directly depend on TagsOfImmutableItemsToReset which will become the
572 // TagsOfImmutableItemsToReset on the next invocation of the function. The
573 // recursion terminates when TagsOfImmutableItemsToReset becomes an empty list
574 // (using the function overload inlined above in the definition of DataBox).

576 template <typename... TagsOfImmutableItemsToReset>

579  tmpl::list<TagsOfImmutableItemsToReset...> /*meta*/) noexcept {
580  EXPAND_PACK_LEFT_TO_RIGHT(reset_compute_item<TagsOfImmutableItemsToReset>());
581  using current_tags_to_reset = tmpl::list<TagsOfImmutableItemsToReset...>;
582  using next_compute_tags_to_reset = tmpl::list_difference<
583  tmpl::remove_duplicates<tmpl::transform<
584  tmpl::append<
585  tmpl::filter<typename DataBox<tmpl::list<Tags...>>::edge_list,
586  std::is_same<tmpl::pin<TagsOfImmutableItemsToReset>,
587  tmpl::get_source<tmpl::_1>>>...>,
588  tmpl::get_destination<tmpl::_1>>>,
589  current_tags_to_reset>;
590  reset_compute_items_after_mutate(next_compute_tags_to_reset{});
591 }

609 /*!
610  * \ingroup DataBoxGroup
611  * \brief Allows changing the state of one or more non-computed elements in
612  * the DataBox
613  *
614  * `mutate()`'s first argument is the DataBox from which to retrieve the tags
615  * `MutateTags`. The objects corresponding to the `MutateTags` are then passed
616  * to `invokable`, which is a lambda or a function object taking as many
617  * arguments as there are `MutateTags` and with the arguments being of types
618  * `gsl::not_null<db::item_type<MutateTags>*>...`. Inside the `invokable` no
619  * items can be retrieved from the DataBox `box`. This is to avoid confusing
620  * subtleties with order of evaluation of compute items, as well as dangling
621  * references. If an `invokable` needs read access to items in `box` they should
622  * be passed as additional arguments to `mutate`. Capturing them by reference in
623  * a lambda does not work because of a bug in GCC 6.3 and earlier. For a
624  * function object the read-only items can also be stored as const references
625  * inside the object by passing `db::get<TAG>(t)` to the constructor.
626  *
627  * \example
628  * \snippet Test_DataBox.cpp databox_mutate_example
629  *
630  * The `invokable` may have function return values, and any returns are
631  * forwarded as returns to the `db::mutate` call.
632  *
633  * \warning Using `db::mutate` returns to obtain non-const references or
634  * pointers to box items is potentially very dangerous. The \ref DataBoxGroup
635  * "DataBox" cannot track any subsequent changes to quantities that have been
636  * "unsafely" extracted in this manner, so we consider it undefined behavior to
637  * return pointers or references to \ref DataBoxGroup "DataBox" contents.
638  */
639 template <typename... MutateTags, typename TagList, typename Invokable,
640  typename... Args>
641 decltype(auto) mutate(const gsl::not_null<DataBox<TagList>*> box,
642  Invokable&& invokable, Args&&... args) noexcept {
643  static_assert(
644  tmpl2::flat_all_v<
645  detail::has_unique_matching_tag_v<TagList, MutateTags>...>,
646  "One of the tags being mutated could not be found in the DataBox or "
647  "is a base tag identifying more than one tag.");
648  static_assert(tmpl2::flat_all_v<tmpl::list_contains_v<
649  typename DataBox<TagList>::mutable_item_tags,
650  detail::first_matching_tag<TagList, MutateTags>>...>,
651  "Can only mutate mutable items");
652  if (UNLIKELY(box->mutate_locked_box_)) {
653  ERROR(
654  "Unable to mutate a DataBox that is already being mutated. This "
655  "error occurs when mutating a DataBox from inside the invokable "
656  "passed to the mutate function.");
657  }
658  box->mutate_locked_box_ = true;
659  using mutate_tags_list =
660  tmpl::list<detail::first_matching_tag<TagList, MutateTags>...>;
661  // For all the tags in the DataBox, check if one of their subtags is
662  // being mutated and if so add the parent to the list of tags
663  // being mutated. Then, remove any tags that would be passed
664  // multiple times.
665  using extra_mutated_tags = tmpl::list_difference<
666  tmpl::filter<TagList,
667  tmpl::bind<tmpl::found, Subitems<tmpl::_1>,
668  tmpl::pin<tmpl::bind<tmpl::list_contains,
669  tmpl::pin<mutate_tags_list>,
670  tmpl::_1>>>>,
671  mutate_tags_list>;
672  // Extract the subtags inside the MutateTags and reset compute items
673  // depending on those too.
674  using full_mutated_items =
675  tmpl::append<detail::expand_subitems<mutate_tags_list>,
676  extra_mutated_tags>;
677 
678  using first_compute_items_to_reset = tmpl::remove_duplicates<
679  tmpl::transform<tmpl::filter<typename DataBox<TagList>::edge_list,
680  tmpl::bind<tmpl::list_contains,
681  tmpl::pin<full_mutated_items>,
682  tmpl::get_source<tmpl::_1>>>,
683  tmpl::get_destination<tmpl::_1>>>;
684  if constexpr (not std::is_same_v<
685  decltype(invokable(
686  make_not_null(
687  &box->template get_item<detail::first_matching_tag<
688  TagList, MutateTags>>()
689  .mutate())...,
690  std::forward<Args>(args)...)),
691  void>) {
692  decltype(auto) return_value = invokable(
693  make_not_null(&box->template get_item<
694  detail::first_matching_tag<TagList, MutateTags>>()
695  .mutate())...,
696  std::forward<Args>(args)...);
697 
698  EXPAND_PACK_LEFT_TO_RIGHT(box->template mutate_mutable_subitems<MutateTags>(
699  typename Subitems<MutateTags>::type{}));
700  box->template reset_compute_items_after_mutate(
701  first_compute_items_to_reset{});
702 
703  box->mutate_locked_box_ = false;
704  return return_value;
705  } else {
706  invokable(
707  make_not_null(&box->template get_item<
708  detail::first_matching_tag<TagList, MutateTags>>()
709  .mutate())...,
710  std::forward<Args>(args)...);
711 
712  EXPAND_PACK_LEFT_TO_RIGHT(box->template mutate_mutable_subitems<MutateTags>(
713  typename Subitems<MutateTags>::type{}));
714  box->template reset_compute_items_after_mutate(
715  first_compute_items_to_reset{});
716 
717  box->mutate_locked_box_ = false;
718  }

1121  * `A1` and `a2` of type `A2`, then

1129  * `sizeof...(Tags)` arguments of types `db::const_item_type<Tags>...`, and
1130  * `sizeof...(Args)` arguments of types `Args...`,
1131  * \code
1132  * result = func(box, db::get<Tags>(box)..., args...);
1133  * \endcode
