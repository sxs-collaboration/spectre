87 template <size_t Dim>
88 struct Metavariables {
89  static constexpr size_t volume_dim = Dim;
90  using system = Elasticity::FirstOrderSystem<Dim>;
91 
92  // List the possible backgrounds, i.e. the variable-independent part of the
93  // equations that define the problem to solve (along with the boundary
94  // conditions). We currently only have analytic solutions implemented, but
95  // will add non-solution backgrounds ASAP.
96  using analytic_solution_registrars = tmpl::flatten<tmpl::list<
97  tmpl::conditional_t<Dim == 2, Elasticity::Solutions::Registrars::BentBeam,
98  tmpl::list<>>,
99  tmpl::conditional_t<Dim == 3,
100  Elasticity::Solutions::Registrars::HalfSpaceMirror,
101  tmpl::list<>>>>;
102  using background_registrars = analytic_solution_registrars;
103  using background_tag = elliptic::Tags::Background<
104  Elasticity::AnalyticData::AnalyticData<Dim, background_registrars>>;
105 
106  // List the possible initial guesses
107  using initial_guess_registrars =
108  tmpl::append<tmpl::list<Elasticity::Solutions::Registrars::Zero<Dim>>,
109  analytic_solution_registrars>;
110  using initial_guess_tag = elliptic::Tags::InitialGuess<
111  ::AnalyticData<Dim, initial_guess_registrars>>;
112 
113  static constexpr Options::String help{
114  "Find the solution to a linear elasticity problem."};
115 
116  // These are the fields we solve for
117  using fields_tag = ::Tags::Variables<typename system::primal_fields>;
118  // These are the fixed sources, i.e. the RHS of the equations
119  using fixed_sources_tag = db::add_tag_prefix<::Tags::FixedSource, fields_tag>;
120  // This is the linear operator applied to the fields. We'll only use it to
121  // apply the operator to the initial guess, so an optimization would be to
122  // re-use the `operator_applied_to_vars_tag` below. This optimization needs a
123  // few minor changes to the parallel linear solver algorithm.
124  using operator_applied_to_fields_tag =
125  db::add_tag_prefix<LinearSolver::Tags::OperatorAppliedTo, fields_tag>;
126 
127  // The linear solver algorithm. We must use GMRES since the operator is
128  // not guaranteed to be symmetric. It can be made symmetric by multiplying by
129  // the DG mass matrix.
130  using linear_solver =
131  LinearSolver::gmres::Gmres<Metavariables, fields_tag,
132  SolveElasticity::OptionTags::GmresGroup,
133  false>;
134  using linear_solver_iteration_id =
135  Convergence::Tags::IterationId<typename linear_solver::options_group>;
136  // For the GMRES linear solver we need to apply the DG operator to its
137  // internal "operand" in every iteration of the algorithm.
138  using vars_tag = typename linear_solver::operand_tag;
139  using operator_applied_to_vars_tag =
140  db::add_tag_prefix<LinearSolver::Tags::OperatorAppliedTo, vars_tag>;
141  // We'll buffer the corresponding fluxes in this tag, but won't actually need
142  // to access them outside applying the operator
143  using fluxes_vars_tag =
144  ::Tags::Variables<db::wrap_tags_in<LinearSolver::Tags::Operand,
145  typename system::primal_fluxes>>;
146 
147  using analytic_solution_fields = typename system::primal_fields;
148  using observe_fields = tmpl::append<
149  analytic_solution_fields,
150  tmpl::list<Elasticity::Tags::Strain<volume_dim>,
151  Elasticity::Tags::PotentialEnergyDensity<volume_dim>>>;
152 
153  // Collect all items to store in the cache.
154  using const_global_cache_tags =
155  tmpl::list<background_tag, initial_guess_tag, Tags::EventsAndTriggers>;
156 
157  struct factory_creation
158  : tt::ConformsTo<Options::protocols::FactoryCreation> {
159  using factory_classes = tmpl::map<
160  tmpl::pair<DomainCreator<volume_dim>, domain_creators<volume_dim>>,
161  tmpl::pair<Event,
162  tmpl::flatten<tmpl::list<
163  Events::Completion,
164  dg::Events::field_observations<
165  volume_dim, linear_solver_iteration_id,
166  observe_fields, analytic_solution_fields>,
167  dg::Events::ObserveVolumeIntegrals<
168  volume_dim, linear_solver_iteration_id,
169  tmpl::list<Elasticity::Tags::PotentialEnergyDensity<
170  volume_dim>>>>>>,
171  tmpl::pair<Trigger,
172  tmpl::push_back<Triggers::logical_triggers,
173  elliptic::Triggers::EveryNIterations<
174  linear_solver_iteration_id>>>>;
175  };
176 
177  // Collect all reduction tags for observers
178  using observed_reduction_data_tags =
179  observers::collect_reduction_data_tags<tmpl::flatten<tmpl::list<
180  tmpl::at<typename factory_creation::factory_classes, Event>,
181  linear_solver>>>;
182 
183  // Specify all global synchronization points.
184  enum class Phase { Initialization, RegisterWithObserver, Solve, Exit };
185 
186  using initialization_actions = tmpl::list<
187  Actions::SetupDataBox,
188  elliptic::dg::Actions::InitializeDomain<volume_dim>,
189  typename linear_solver::initialize_element,
190  elliptic::Actions::InitializeFields<system, initial_guess_tag>,
191  elliptic::Actions::InitializeFixedSources<system, background_tag>,
192  Initialization::Actions::AddComputeTags<tmpl::list<
193  Elasticity::Tags::ConstitutiveRelationReference<volume_dim,
194  background_tag>,
195  Elasticity::Tags::StrainCompute<volume_dim>,
196  Elasticity::Tags::PotentialEnergyDensityCompute<volume_dim>>>,
197  elliptic::Actions::InitializeOptionalAnalyticSolution<
198  background_tag,
199  tmpl::append<typename system::primal_fields,
200  typename system::primal_fluxes>,
201  Elasticity::Solutions::AnalyticSolution<Dim, background_registrars>>,
202  elliptic::dg::Actions::initialize_operator<system>,
203  elliptic::dg::Actions::ImposeInhomogeneousBoundaryConditionsOnSource<
204  system, fixed_sources_tag>,
205  // Apply the DG operator to the initial guess
206  elliptic::dg::Actions::apply_operator<
207  system, true, linear_solver_iteration_id, fields_tag, fluxes_vars_tag,
208  operator_applied_to_fields_tag, vars_tag, fluxes_vars_tag>,
209  Initialization::Actions::RemoveOptionsAndTerminatePhase>;
210 
211  using build_linear_operator_actions = elliptic::dg::Actions::apply_operator<
212  system, true, linear_solver_iteration_id, vars_tag, fluxes_vars_tag,
213  operator_applied_to_vars_tag>;
214 
215  using register_actions =
216  tmpl::list<observers::Actions::RegisterEventsWithObservers,
217  Parallel::Actions::TerminatePhase>;
218 
219  using solve_actions = tmpl::list<
220  typename linear_solver::template solve<tmpl::list<
221  Actions::RunEventsAndTriggers, build_linear_operator_actions>>,
222  Actions::RunEventsAndTriggers, Parallel::Actions::TerminatePhase>;
223 
224  using dg_element_array = elliptic::DgElementArray<
225  Metavariables,
226  tmpl::list<Parallel::PhaseActions<Phase, Phase::Initialization,
227  initialization_actions>,
228  Parallel::PhaseActions<Phase, Phase::RegisterWithObserver,
229  register_actions>,
230  Parallel::PhaseActions<Phase, Phase::Solve, solve_actions>>>;
231 
232  // Specify all parallel components that will execute actions at some point.
233  using component_list = tmpl::flatten<
234  tmpl::list<dg_element_array, typename linear_solver::component_list,
235  observers::Observer<Metavariables>,
236  observers::ObserverWriter<Metavariables>>>;

238  // Specify the transitions between phases.
239  template <typename... Tags>
240  static Phase determine_next_phase(
241  const gsl::not_null<
242  tuples::TaggedTuple<Tags...>*> /*phase_change_decision_data*/,
243  const Phase& current_phase,
244  const Parallel::CProxy_GlobalCache<
245  Metavariables>& /*cache_proxy*/) noexcept {
246  switch (current_phase) {
247  case Phase::Initialization:
248  return Phase::RegisterWithObserver;
249  case Phase::RegisterWithObserver:
250  return Phase::Solve;
251  case Phase::Solve:
252  return Phase::Exit;
253  case Phase::Exit:
254  ERROR(
255  "Should never call determine_next_phase with the current phase "
256  "being 'Exit'");
257  default:
258  ERROR(
259  "Unknown type of phase. Did you static_cast<Phase> an integral "
260  "value?");
261  }
262  }
