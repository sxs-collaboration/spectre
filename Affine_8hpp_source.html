25 /*!
26  * \ingroup CoordinateMapsGroup
27  * \brief Affine map from \f$\xi \in [A, B]\rightarrow x \in [a, b]\f$.
28  *
29  * The formula for the mapping is...
30  * \f[
31  * x = \frac{b}{B-A} (\xi-A) +\frac{a}{B-A}(B-\xi)
32  * \f]
33  * \f[
34  * \xi =\frac{B}{b-a} (x-a) +\frac{A}{b-a}(b-x)
35  * \f]
36  */
