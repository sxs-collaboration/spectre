67 /*!
68  * \ingroup DiscontinuousGalerkinGroup
69  * \brief %Observe volume tensor fields.
70  *
71  * A class that writes volume quantities to an h5 file during the simulation.
72  * The observed quantitites are:
73  * - `InertialCoordinates`
74  * - Tensors listed in `Tensors` template parameter
75  * - `Error(*)` = errors in `AnalyticSolutionTensors` =
76  * \f$\text{value} - \text{analytic solution}\f$
77  *
78  * The user may specify an `interpolation_mesh` to which the
79  * data is interpolated.
80  */

222  // We factor out the work into a static member function so it can be shared
223  // with other field observing events, like the one that deals with DG-subcell
224  // where there are two grids. This is to avoid copy-pasting all of the code.
225  template <typename OptionalAnalyticSolutions, typename Metavariables,
226  typename ParallelComponent>
227  static void call_operator_impl(
228  const std::string& subfile_path,
229  const std::unordered_map<std::string, FloatingPointType>&
230  variables_to_observe,
231  const std::optional<Mesh<VolumeDim>>& interpolation_mesh,
232  const typename ObservationValueTag::type& observation_value,
233  const Mesh<VolumeDim>& mesh,
234  const tnsr::I<DataVector, VolumeDim, Frame::Inertial>&
235  inertial_coordinates,
236  const typename AnalyticSolutionTensors::
237  type&... analytic_solution_tensors,
238  const typename NonSolutionTensors::type&... non_solution_tensors,
239  const OptionalAnalyticSolutions& optional_analytic_solutions,
240  Parallel::GlobalCache<Metavariables>& cache,
241  const ElementId<VolumeDim>& array_index,
242  const ParallelComponent* const /*meta*/) noexcept {
243  const auto analytic_solutions = [&optional_analytic_solutions]() {
244  if constexpr (tt::is_a_v<std::optional, OptionalAnalyticSolutions>) {
245  return optional_analytic_solutions.has_value()
246  ? std::make_optional(std::cref(*optional_analytic_solutions))
247  : std::nullopt;
248  } else {
249  return std::make_optional(std::cref(optional_analytic_solutions));
250  }
251  }();
252 
253  const std::string element_name =
254  MakeString{} << ElementId<VolumeDim>(array_index) << '/';
255 
256  // if no interpolation_mesh is provided, the interpolation is essentially
257  // ignored by the RegularGridInterpolant except for a single copy.
258  const intrp::RegularGrid interpolant(mesh,
259  interpolation_mesh.value_or(mesh));
260 
261  // Remove tensor types, only storing individual components.
262  std::vector<TensorComponent> components;
263  // This is larger than we need if we are only observing some
264  // tensors, but that's not a big deal and calculating the correct
265  // size is nontrivial.
266  components.reserve(alg::accumulate(
267  std::initializer_list<size_t>{
268  inertial_coordinates.size(),
269  (analytic_solutions.has_value() ? 2 : 1) *
270  AnalyticSolutionTensors::type::size()...,
271  NonSolutionTensors::type::size()...},
272  0_st));
273 
274  const auto record_tensor_components = [&components, &element_name,
275  &interpolant, &variables_to_observe](
276  const auto tensor_tag_v,
277  const auto& tensor) noexcept {
278  using tensor_tag = tmpl::type_from<decltype(tensor_tag_v)>;
279  if (variables_to_observe.count(db::tag_name<tensor_tag>()) == 1) {
280  const auto floating_point_type =
281  variables_to_observe.at(db::tag_name<tensor_tag>());
282  for (size_t i = 0; i < tensor.size(); ++i) {
283  const auto tensor_component = interpolant.interpolate(tensor[i]);
284  if (floating_point_type == FloatingPointType::Float) {
285  components.emplace_back(element_name + db::tag_name<tensor_tag>() +
286  tensor.component_suffix(i),
287  std::vector<float>{tensor_component.begin(),
288  tensor_component.end()});
289  } else {
290  components.emplace_back(element_name + db::tag_name<tensor_tag>() +
291  tensor.component_suffix(i),
292  tensor_component);
293  }
294  }
295  }
296  };
297  record_tensor_components(tmpl::type_<coordinates_tag>{},
298  inertial_coordinates);
299  EXPAND_PACK_LEFT_TO_RIGHT(record_tensor_components(
300  tmpl::type_<AnalyticSolutionTensors>{}, analytic_solution_tensors));
301  EXPAND_PACK_LEFT_TO_RIGHT(record_tensor_components(
302  tmpl::type_<NonSolutionTensors>{}, non_solution_tensors));
303 
304  if (analytic_solutions.has_value()) {
305  const auto record_errors =
306  [&analytic_solutions, &components, &element_name, &interpolant,
307  &variables_to_observe](const auto tensor_tag_v,
308  const auto& tensor) noexcept {
309  using tensor_tag = tmpl::type_from<decltype(tensor_tag_v)>;
310  if (variables_to_observe.count(db::tag_name<tensor_tag>()) == 1) {
311  const auto floating_point_type =
312  variables_to_observe.at(db::tag_name<tensor_tag>());
313  for (size_t i = 0; i < tensor.size(); ++i) {
314  DataVector error = interpolant.interpolate(
315  DataVector(tensor[i] - get<::Tags::Analytic<tensor_tag>>(
316  analytic_solutions->get())[i]));
317  if (floating_point_type == FloatingPointType::Float) {
318  components.emplace_back(
319  element_name + "Error(" + db::tag_name<tensor_tag>() +
320  ")" + tensor.component_suffix(i),
321  std::vector<float>{error.begin(), error.end()});
322  } else {
323  components.emplace_back(element_name + "Error(" +
324  db::tag_name<tensor_tag>() + ")" +
325  tensor.component_suffix(i),
326  std::move(error));
327  }
328  }
329  }
330  };
331  EXPAND_PACK_LEFT_TO_RIGHT(record_errors(
332  tmpl::type_<AnalyticSolutionTensors>{}, analytic_solution_tensors));
333 
334  (void)(record_errors); // Silence GCC warning about unused variable
335  }

352  }

386 template <size_t VolumeDim, typename ObservationValueTag, typename... Tensors,
387  typename... AnalyticSolutionTensors, typename... NonSolutionTensors>
388 ObserveFields<VolumeDim, ObservationValueTag, tmpl::list<Tensors...>,
389  tmpl::list<AnalyticSolutionTensors...>,
390  tmpl::list<NonSolutionTensors...>>::
391  ObserveFields(const std::string& subfile_name,
392  const FloatingPointType coordinates_floating_point_type,
393  const std::vector<FloatingPointType>& floating_point_types,
394  const std::vector<std::string>& variables_to_observe,
395  std::optional<Mesh<VolumeDim>> interpolation_mesh,
396  const Options::Context& context)
397  : subfile_path_("/" + subfile_name),
398  variables_to_observe_([&context, &floating_point_types,
399  &variables_to_observe]() {
400  if (floating_point_types.size() != 1 and
401  floating_point_types.size() != variables_to_observe.size()) {
402  PARSE_ERROR(context, "The number of floating point types specified ("
403  << floating_point_types.size()
404  << ") must be 1 or the number of variables "
405  "specified for observing ("
406  << variables_to_observe.size() << ")");
407  }
408  std::unordered_map<std::string, FloatingPointType> result{};
409  for (size_t i = 0; i < variables_to_observe.size(); ++i) {
410  result[variables_to_observe[i]] = floating_point_types.size() == 1
411  ? floating_point_types[0]
412  : floating_point_types[i];
413  ASSERT(
414  result.at(variables_to_observe[i]) == FloatingPointType::Float or
415  result.at(variables_to_observe[i]) ==
416  FloatingPointType::Double,
417  "Floating point type for variable '"
418  << variables_to_observe[i]
419  << "' must be either Float or Double.");
420  }
421  return result;
422  }()),
423  interpolation_mesh_(interpolation_mesh) {
424  using ::operator<<;
425  const std::unordered_set<std::string> valid_tensors{
426  db::tag_name<Tensors>()...};
427  for (const auto& [name, floating_point_type] : variables_to_observe_) {
428  (void)floating_point_type;
429  if (valid_tensors.count(name) != 1) {
430  PARSE_ERROR(
431  context,
432  name << " is not an available variable. Available variables:\n"
433  << (std::vector<std::string>{db::tag_name<Tensors>()...}));
434  }
435  if (alg::count(variables_to_observe, name) != 1) {
436  PARSE_ERROR(context, name << " specified multiple times");
437  }
438  }
439  variables_to_observe_[coordinates_tag::name()] =
440  coordinates_floating_point_type;
441 }
